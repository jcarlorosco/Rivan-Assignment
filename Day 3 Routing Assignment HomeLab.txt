1. Assign IP Address on P1, P2, S1, S2
	P1 = 10.2.1.101 /24
	P2 = 10.2.1.102 /24
	S1 = 192.168.1.133 /27
	S2 = 10.2.2.133 /24

!@P1
conf t
 int e0/0
 no shut
 ip add 10.2.1.101 255.255.255.0
end

!@P2
conf t
 int e1/0
 no shut
 ip add 10.2.1.102 255.255.255.0
end

!@S1
conf t
 int e1/0
 no shut
 ip add 192.168.1.133 255.255.255.224
end

!@S2
conf t
 int e1/0
 no shut
 ip add 10.2.2.133 255.255.255.0
end

!@CLI Commands to take note
sh ip int br (Show IP interface brief)

2. Assign them to the correct VLANs on their appropriate access switch A1, A2, D1, D2

!@A1
conf t
 int e0/0
  switchport mode access
  switch port access vlan 10

!@A2
conf t
 int e1/0
  switchport mode access
  switch port access vlan 10
end

!@D1
conf t
 int e1/0
  switchport mode access
  switch port access vlan 100
end

!@D2
conf t
 int e1/0
  switch port mode access
  switch port access vlan 20
end

!@check if  they can ping each other

!@CLI Commands to take note
do sh hist (Do show history - show recent commands)
sh vlan br (Show VLAN Brief)

5. OSPF

!@R4
conf t
 int e1/2
  ip  ospf network point-to-point
  no shut
  exit
 router ospf 1
  router-id 4.4.4.4
  network 4.4.4.4 0.0.0.0 area 0
  network 10.1.1.8 0.0.0.3 area 0
end

!@R3
conf t
  int e1/2
  ip  ospf network point-to-point
 no shut
  exit
 router ospf 1
  router-id 3.3.3.3
  network 3.3.3.3 0.0.0.0 area 0
  network 10.1.1.8 0.0.0.3 area 0
  network 10.1.1.4 0.0.0.3 area 0
end

!@R2
conf t
  int e1/2
   ip ospf priority 255
  no shut
  exit
 router ospf 1
  router-id 2.2.2.2
  network 2.2.2.2 0.0.0.0 area 0
  network 10.1.1.4 0.0.0.3 area 0
  network 10.1.1.0 0.0.0.3 area 0
end

!@R1
conf t
  int e1/0
  ip ospf priority 0
  no shut
  exit
 router ospf 1
  router-id 1.1.1.1
  network 1.1.1.1 0.0.0.0 area 0
  network 10.1.1.0 0.0.0.3 area 0
end


!@Check ospf neighbor
show ip ospf neighbor

!@Reset OSPF
clear ip ospf process
